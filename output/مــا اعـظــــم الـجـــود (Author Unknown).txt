مــا أعـظــــمَ الـجـــود الـــــذى ذخــرتـه
لـــخـــائــفــيـك و غـنـاك وهــبــتــه
رحــمــتـُـــك تــــحـــيـــط ُ بــى
و بـتـرنـْم النجاة دوما تـكـتــنـفــنـى
Ma adamal joodal lathy dakhartahu likha'ifika wa ghinak wahabtahu rahmatuka tuheetu bee wa bitaranum ilnajaty dowman taktanifuny

أبــتــهــجُ و أفــــرحُ
قد جعــــل عجـــــبــــاً رحــــمــــته لـــى
فهو صــخرة ٌ و مـــــــلجأ ٌ
أ عـــــظـمُ مـن قـــــد فــدى نفـــــــســـــى
Abtahiju wa afrahou kad ja'ala ajaban rahmatu lee fahwa sakhratun wa malja'un u'athimu man kad fada nafsee

الـربُ نـورى و خـلاصــى لن أخـــــــافْ
 إن نـز ل عــلـىّ جـيـــشٌ لا أخـــــــــــافْ
الآن يـرتـفعُ رأسـى عـلى أعـدائى من حــــولـــى
وللــربِ أرنـــــــــــمُ
Arrabu nury wa khalasy lan akhaf in nazala alaya jayshun la akhaf al'ana yartafi'u ra'sy ala ada'ee min hawly wa lirraby uranimu
